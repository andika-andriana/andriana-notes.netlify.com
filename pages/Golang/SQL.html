        return

        if err != nil {

            return

    }

        fmt.Println(err.Error())
        return
    }

    for _, each := range result {
        fmt.Println(each.name)
    }
}

  1. Membaca 1 Record Data Menggunakan MethodÂ QueryRow()

    Untuk query yang menghasilkan 1 baris record saja, bisa gunakan methodÂ QueryRow(), dengan metode ini kode menjadi lebih ringkas. Chain dengan methodÂ Scan()Â untuk mendapatkan value-nya.

    package main

import "fmt"
import "database/sql"
import _ "mysql-master"

type student struct {
	id    string
	name  string
	age   int
	grade int
}

func connect() (*sql.DB, error) {
	db, err := sql.Open("mysql", "root:@tcp(127.0.0.1:3306)/db_belajar_golang")
	if err != nil {
		return nil, err
	}
	return db, nil
}

func sqlQueryRow() {
	var db, err = connect()
	if err != nil {
		fmt.Println(err.Error())
		return
	}
	defer db.Close()

	var result = student{}
	var id = "E001"
	err = db.
		QueryRow("select name, grade from tb_student where id = ?", id).
		Scan(&result.name, &result.grade)
	if err != nil {
		fmt.Println(err.Error())
		return
	}

	fmt.Printf("name: %s\ngrade: %d\n", result.name, result.grade)
}

func main() {
	sqlQueryRow()
}

    Dari kode di atas ada statement yang dituliskan cukup unik, chain statement boleh dituliskan dalam beberapa baris, contohnya:

    err = db.
    QueryRow("select name, grade from tb_student where id = ?", id).

  1. Eksekusi Query MenggunakanÂ Prepare()

    TeknikÂ prepared statementÂ adalah teknik penulisan query di awal dengan kelebihan bisa di re-use atau digunakan banyak kali untuk eksekusi yang berbeda-beda.

    Metode ini bisa digabung denganÂ Query()Â maupunÂ QueryRow(). Berikut merupakan contoh penerapannya.

    package main

import "fmt"
import "database/sql"
import _ "mysql-master"

type student struct {
	id    string
	name  string
	age   int
	grade int
}

func connect() (*sql.DB, error) {
	db, err := sql.Open("mysql", "root:@tcp(127.0.0.1:3306)/db_belajar_golang")
	if err != nil {
		return nil, err
	}
	return db, nil
}

func sqlPrepare() {
	db, err := connect()
	if err != nil {
		fmt.Println(err.Error())
		return
	}
	defer db.Close()

	stmt, err := db.Prepare("select name, grade from tb_student where id = ?")
	if err != nil {
		fmt.Println(err.Error())
		return
	}

	var result1 = student{}
	stmt.QueryRow("E001").Scan(&result1.name, &result1.grade)
	fmt.Printf("name: %s\ngrade: %d\n", result1.name, result1.grade)

	var result2 = student{}
	stmt.QueryRow("W001").Scan(&result2.name, &result2.grade)
	fmt.Printf("name: %s\ngrade: %d\n", result2.name, result2.grade)

	var result3 = student{}
	stmt.QueryRow("B001").Scan(&result3.name, &result3.grade)
	fmt.Printf("name: %s\ngrade: %d\n", result3.name, result3.grade)
}

func main() {
	sqlPrepare()
}

    MethodÂ Prepare()Â digunakan untuk deklarasi query, yang mengembalikan objek bertipeÂ sql.*Stmt. Dari objek tersebut, dipanggil methodÂ QueryRow()Â beberapa kali dengan isi value untukÂ idÂ berbeda-beda untuk tiap pemanggilannya.

  1. Insert, Update, & Delete Data MenggunakanÂ Exec()

    Untuk operasiÂ insert,Â update, danÂ delete; dianjurkan untuk tidak menggunakan fungsiÂ sql.Query()Â ataupunÂ sql.QueryRow()Â untuk eksekusinya. Direkomendasikan eksekusi perintah-perintah tersebut lewat fungsiÂ Exec(), contohnya seperti pada kode berikut.

    package main

import "fmt"
import "database/sql"
import _ "mysql-master"

type student struct {
	id    string
	name  string
	age   int
	grade int
}

func connect() (*sql.DB, error) {
	db, err := sql.Open("mysql", "root:@tcp(127.0.0.1:3306)/db_belajar_golang")
	if err != nil {
		return nil, err
	}
	return db, nil
}

func sqlExec() {
	db, err := connect()
	if err != nil {
		fmt.Println(err.Error())
		return
	}
	defer db.Close()

	_, err = db.Exec("insert into tb_student values (?, ?, ?, ?)", "G001", "Galahad", 29, 2)
	if err != nil {
		fmt.Println(err.Error())
		return
	}
	fmt.Println("insert success!")

	_, err = db.Exec("update tb_student set age = ? where id = ?", 28, "G001")
	if err != nil {
		fmt.Println(err.Error())
		return
	}
	fmt.Println("update success!")

	_, err = db.Exec("delete from tb_student where id = ?", "G001")
	if err != nil {
		fmt.Println(err.Error())
		return
	}
	fmt.Println("delete success!")
}

func main() {
	sqlExec()
}

    Teknik prepared statement juga bisa digunakan pada metode ini.

    Berikut adalah perbandingan eksekusiÂ Exec()Â menggunakanÂ Prepare()Â dan cara biasa.

  1. Koneksi Dengan Engine Database Lain

    Karena packageÂ database/sqlÂ merupakan interface generic, maka cara untuk koneksi ke engine database lain (semisal Oracle, Postgres, SQL Server) adalah sama dengan cara koneksi ke MySQL.     Sebagai contoh saya menggunakan driverÂ pqÂ untuk koneksi ke server Postgres, maka connection string-nya:
