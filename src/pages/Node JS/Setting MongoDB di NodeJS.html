  1. Buka project pada text editor kita (contoh: atom.io) pada index.js import package mongoose dan body-parser:
    // filename: index.js

// import express
let express = require("express");

// import API Router
let apiRoutes = require("./apiRoutes");

// import mongoose
let mongoose = require("mongoose");

// import body-parser
let bodyParser = require("body-parser");

// initialize app
let app = express();

// setup server
var port = process.env.PORT || 8080;

// send message for default URL
app.get("/", (req, res) => res.send("Hello Dunia with Express dan Nodemon!"));

// app apiRoutes
app.use("/api", apiRoutes);

// launch app to listen specified PORT
app.listen(port, function() {
  console.log("Running resthub on port: " + port);
});
  1. Kemudian lakukan configurasi body-parser:
    // filename: index.js

// import express
let express = require("express");

// import API Router
let apiRoutes = require("./apiRoutes");

// import mongoose
let mongoose = require("mongoose");

// import body-parser
let bodyParser = require("body-parser");

// initialize app
let app = express();

// setup server
var port = process.env.PORT || 8080;

// configuration body-parser
app.use(
  bodyParser.urlencoded({
    extended: true
  })
);
app.use(bodyParser.json());

// send message for default URL
app.get("/", (req, res) => res.send("Hello Dunia with Express dan Nodemon!"));

// app apiRoutes
app.use("/api", apiRoutes);

// launch app to listen specified PORT
app.listen(port, function() {
  console.log("Running resthub on port: " + port);
});
  1. Kemudian hubungkan project dengan mongoose dengan menuliskan kode didalam index.js:
    // filename: index.js

// import express
let express = require("express");

// import API Router
let apiRoutes = require("./apiRoutes");

// import mongoose
let mongoose = require("mongoose");

// import body-parser
let bodyParser = require("body-parser");

// initialize app
let app = express();

// setup server
var port = process.env.PORT || 8080;

// configuration body-parser
app.use(
  bodyParser.urlenconded({
    extended: true
  })
);
app.use(bodyParser.json());

// connect mongoose and set connection variable
mongoose.connect("mongodb://localhost/resthub");
var db = mongoose.connection;

// send message for default URL
app.get("/", (req, res) => res.send("Hello Dunia with Express dan Nodemon!"));

// app apiRoutes
app.use("/api", apiRoutes);

// launch app to listen specified PORT
app.listen(port, function() {
  console.log("Running resthub on port: " + port);
});
