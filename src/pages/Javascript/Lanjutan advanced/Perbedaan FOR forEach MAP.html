  • Berikut penjelasan tentang method forEach:
    • Dapat kita gunakan untuk melakukan perulangan berdasarkan panjang dari sebuah array
    function perulanganForEach() {
  var dataNama = ['Andi', 'Bona', 'Bobi', 'Dodi'];
  dataNama.forEach((item, index, array) => {
    console.log(item);
    console.log(index);
    console.log(array);
  });
}
